Struct actix::prelude::System[−][src]

pub struct System { /* fields omitted */ }

System is an actor which manages process.

Before starting any actix's actors, System actor has to be created with System::new() call. This method creates new Arbiter in current thread and starts System actor.

Examples

extern crate actix;

use actix::prelude::*;
use std::time::Duration;

struct Timer {
    dur: Duration,
}

impl Actor for Timer {
    type Context = Context<Self>;

    // stop system after `self.dur` seconds
    fn started(&mut self, ctx: &mut Context<Self>) {
        ctx.run_later(self.dur, |act, ctx| {
            // send `SystemExit` to `System` actor.
            Arbiter::system().do_send(actix::msgs::SystemExit(0));
        });
    }
}

fn main() {
    // initialize system
    let sys = System::new("test");

    // Start `Timer` actor
    let _: () = Timer {
        dur: Duration::new(0, 1),
    }.start();

    // Run system, this function blocks current thread
    let code = sys.run();
    std::process::exit(code);
}
